function updateQuestions() { const text = document.getElementById('questionsTextarea').value; const lines = text.trim().split('\n'); const newQs = lines.map(line => { const parts = line.split('|'); return { q: parts[0], a: parts[1], o: parts[2] ? parts[2].split(',') : [] }; }); // 必須使用與 index.html 相同的 Key localStorage.setItem('flower_advanced_questions', JSON.stringify(newQs)); alert("題庫已更新！請在手機上開啟遊戲確認。"); }
